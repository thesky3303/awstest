cat > /usr/local/bin/ticketing_movie_cache_boot_warm.sh <<'WARMSHEOF'
#!/bin/bash
set -u

BASE_DIR="/mnt/hgfs/ticketing-was"
PYTHON_BIN="/root/ticketing-was/venv/bin/python"
RUNNER_FILE="$BASE_DIR/movie/movie_cache_builder.py"
LOG_FILE="/var/log/ticketing-movie-cache.log"

attempt=1
success=0

while [ "$attempt" -le 5 ]; do
    echo "[$(date '+%F %T')] movie cache warm attempt $attempt using $PYTHON_BIN" >> "$LOG_FILE"

    if cd "$BASE_DIR" && PYTHONPATH="$BASE_DIR" "$PYTHON_BIN" "$RUNNER_FILE" >> "$LOG_FILE" 2>&1; then
        echo "[$(date '+%F %T')] movie cache warm success on attempt $attempt" >> "$LOG_FILE"
        success=1
    else
        echo "[$(date '+%F %T')] movie cache warm failed on attempt $attempt" >> "$LOG_FILE"
    fi

    if [ "$attempt" -lt 5 ]; then
        sleep 60
    fi

    attempt=$((attempt + 1))
done

if [ "$success" -eq 1 ]; then
    exit 0
fi

exit 1
WARMSHEOF

chmod +x /usr/local/bin/ticketing_movie_cache_boot_warm.sh

cat > /etc/systemd/system/ticketing-movie-cache-boot-warm.service <<'WARMSVCEOF'
[Unit]
Description=Warm movie cache during first 5 minutes after boot
After=network-online.target redis.service
Wants=network-online.target redis.service

[Service]
Type=oneshot
WorkingDirectory=/mnt/hgfs/ticketing-was
Environment=PYTHONPATH=/mnt/hgfs/ticketing-was
ExecStart=/usr/local/bin/ticketing_movie_cache_boot_warm.sh
RemainAfterExit=no
User=root

[Install]
WantedBy=multi-user.target
WARMSVCEOF
